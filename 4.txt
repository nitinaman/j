public class calculator {

    public int add(int a, int b) {
        return a + b;
    }

    public int sub(int a, int b) {
        return a - b;
    }

    public int mul(int a, int b) {
        return a * b;
    }

    public int div(int a, int b) {
        if (b == 0)
            throw new ArithmeticException("Cannot divide by zero");
        return a / b;
    }

    public static void main(String[] args) {
        calculator c = new calculator();

        System.out.println("Addition: " + c.add(10, 5));
        System.out.println("Subtraction: " + c.sub(10, 5));
        System.out.println("Multiplication: " + c.mul(10, 5));
        System.out.println("Division: " + c.div(10, 5));
    }
}



import static org.junit.Assert.*;
import org.junit.Test;

public class CalculatorTest {

    calculator c = new calculator();

    @Test
    public void testAdd() {
        assertEquals(15, c.add(10, 5));
    }

    @Test
    public void testSub() {
        assertEquals(5, c.sub(10, 5));
    }

    @Test
    public void testMul() {
        assertEquals(50, c.mul(10, 5));
    }

    @Test
    public void testDiv() {
        assertEquals(2, c.div(10, 5));
    }

    @Test(expected = ArithmeticException.class)
    public void testDivByZero() {
        c.div(10, 0);
    }
}

public class calculator {

	public int search(int[] arr, int key) {
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == key) {
                return i;
            }
        }
        return -1;
    }
}


import static org.junit.Assert.*;
import org.junit.Test;

public class calculator_test {

    calculator ls = new calculator();

    @Test
    public void testFound() {
        int[] arr = {1, 2, 3, 4};
        assertEquals(2, ls.search(arr, 3));
    }

    @Test
    public void testNotFound() {
        int[] arr = {1, 2, 3, 4};
        assertEquals(-1, ls.search(arr, 5));
    }
}

public class BankAccount {
	private double balance;

    public BankAccount(double initialBalance) {
        this.balance = initialBalance;
    }

    public double getBalance() {
        return balance;
    }

    public void deposit(double amount) {
        if (amount <= 0) throw new IllegalArgumentException("Invalid deposit");
        balance += amount;
    }

    public void withdraw(double amount) {
        if (amount <= 0 || amount > balance) {
            throw new IllegalArgumentException("Invalid withdrawal");
        }
        balance -= amount;
    }
}


import static org.junit.Assert.*;
import org.junit.Test;

public class calculator_test {

    @Test
    public void testDeposit() {
        BankAccount acc = new BankAccount(100);
        acc.deposit(50);
        assertEquals(150, acc.getBalance());
    }

    @Test
    public void testWithdraw() {
        BankAccount acc = new BankAccount(100);
        acc.withdraw(40);
        assertEquals(60, acc.getBalance());
    }

    @Test(expected = IllegalArgumentException.class)
    public void testInvalidWithdraw() {
        BankAccount acc = new BankAccount(100);
        acc.withdraw(200);
    }
}
